018/**
019 * Ein {@code GameState} beinhaltet alle Informationen, die den Spielstand zu
020 * einem gegebenen Zeitpunkt, das heisst zwischen zwei Spielzuegen, beschreiben.
021 * Dies umfasst eine fortlaufende Zugnummer ({@link #getTurn() getRound()}), die
022 * der Spielserver als Antwort von einem der beiden Spieler (
023 * {@link #getCurrentPlayer() getCurrentPlayer()}) erwartet. Weiterhin gehoeren
024 * die Informationen ueber die beiden Spieler und das Spielfeld zum Zustand.
025 * Zuseatzlich wird ueber den zuletzt getaetigeten Spielzung und ggf. ueber das
026 * Spielende informiert.
027 *
028 *
029 * Der {@code GameState} ist damit das zentrale Objekt ueber das auf alle
030 * wesentlichen Informationen des aktuellen Spiels zugegriffen werden kann.
031 *
032 *
033 * Der Spielserver sendet an beide teilnehmenden Spieler nach jedem getaetigten
034 * Zug eine neue Kopie des {@code GameState}, in dem der dann aktuelle Zustand
035 * beschrieben wird. Informationen ueber den Spielverlauf sind nur bedingt ueber
036 * den {@code GameState} erfragbar und muessen von einem Spielclient daher bei
037 * Bedarf selbst mitgeschrieben werden.
038 *
039 *
040 * Zusaetzlich zu den eigentlichen Informationen koennen bestimmte
041 * Teilinformationen abgefragt werden.
042 *
043 * @author Niklas, Sören
044 */
045@XStreamAlias(value = "state")
046public class GameState implements Cloneable {

088  /**
089   * Erzeugt einen neuen {@code GameState}, in dem alle Informationen so gesetzt
090   * sind, wie sie zu Beginn eines Spiels, bevor die Spieler beigetreten sind,
091   * gueltig sind.
092   *
093   *
094   * Dieser Konstruktor ist nur fuer den Spielserver relevant und sollte vom
095   * Spielclient i.A. nicht aufgerufen werden!
096   *
097   * Das Spielfeld wird zufällig aufgebaut.

099  public GameState() {
100    this.turn = 0;
101    this.currentPlayer = PlayerColor.RED;
102    this.startPlayer = PlayerColor.RED;
103    this.board = new Board();
104    this.red = new Player(PlayerColor.RED);
105    this.blue = new Player(PlayerColor.BLUE);
106  }

108  /**
109   * Erzeugt einen neuen {@code GameState} mit denselben Eigenschaften von
110   * stateToClone. Fuer eigene Implementierungen.
111   */
112  protected GameState(GameState stateToClone) throws CloneNotSupportedException {
113    GameState clone = stateToClone.clone();
114    setRedPlayer(clone.getPlayer(PlayerColor.RED));
115    setBluePlayer(clone.getPlayer(PlayerColor.BLUE));
116    setLastMove(clone.getLastMove());
117    setBoard(clone.getBoard());
118    setCurrentPlayer(clone.getCurrentPlayerColor());
119  }

121  /**
122   * erzeugt eine Deepcopy dieses Objekts
123   *
124   * @return ein neues Objekt mit gleichen Eigenschaften
125   * @throws CloneNotSupportedException falls das Klonen fehlschlaegt
126   */
127  @Override
128  public GameState clone() throws CloneNotSupportedException {
129    GameState clone = (GameState) super.clone();
130    if (red != null)
131      clone.red = this.red.clone();
132    if (blue != null)
133      clone.blue = this.blue.clone();
134    if (lastMove != null)
135      clone.lastMove = (Move) this.lastMove.clone();
136    if (board != null)
137      clone.board = this.board.clone();
138    if (currentPlayer != null)
139      clone.currentPlayer = this.currentPlayer;
140    clone.startPlayer = this.startPlayer;
141    clone.turn = this.turn;
142    return clone;
143  }

295   * Überprüft ob ein Feld durch einen Spieler belegt ist, sodass niemand darauf ziehen kann.

308  /**
309   * Überprüft ob der angegebene Spieler an erster Stelle ist. Wenn sich beide
310   * Spieler im Ziel befinden wird zusätzlich überprüft, ob <code>player</code>
311   * weniger Karotten besitzt als der Gegenspieler.
312   * 
313   * @param player überprüfter Spieler
314   * @return true, falls Spieler an erster Stelle
315   */
316  public final boolean isFirst(final Player player)
317  {
318    Player o = this.getOpponent(player);
319    boolean isFirst = o.getFieldIndex() <= player.getFieldIndex();
320    if (player.inGoal() && o.getFieldIndex() == player.getFieldIndex())
321      isFirst = isFirst
322          && player.getCarrots() < o.getCarrots();
323    return isFirst;
324  }

434   * {@link #getPlayerStats(PlayerColor) Spielerstats}, wobei getGameStats()[0],
435   * einem Aufruf von getPlayerStats(PlayerColor.RED) entspricht.
436   *
437   * @see #getPlayerStats(PlayerColor)
438   * @return Statusinformationen beider Spieler
439   */
440  public int[][] getGameStats() {
441
442    int[][] stats = new int[2][2];
443
444    stats[Constants.GAME_STATS_RED_INDEX][Constants.GAME_STATS_FIELD_INDEX] = this.red.getFieldIndex();
445    stats[Constants.GAME_STATS_RED_INDEX][Constants.GAME_STATS_CARROTS] = this.red.getCarrots();
446    stats[Constants.GAME_STATS_BLUE_INDEX][Constants.GAME_STATS_FIELD_INDEX] = this.blue.getFieldIndex();
447    stats[Constants.GAME_STATS_BLUE_INDEX][Constants.GAME_STATS_CARROTS] = this.blue.getCarrots();
448    return stats;

544  public ArrayList<Move> getPossibleMoves() {
545    ArrayList<Move> possibleMove = new ArrayList<>();
546    ArrayList<Action> actions = new ArrayList<>();
547    if (GameRuleLogic.isValidToEat(this)) {
548      // Wenn ein Salat gegessen werden kann, muss auch ein Salat gegessen werden
549      actions.add(new EatSalad());
550      Move move = new Move(actions);
551      possibleMove.add(move);
552      return possibleMove;
553    }
554    if (GameRuleLogic.isValidToExchangeCarrots(this, 10)) {
555      actions.add(new ExchangeCarrots(10));
556      possibleMove.add(new Move(actions));
557      actions.clear();
558    }
559    if (GameRuleLogic.isValidToExchangeCarrots(this, -10)) {
560      actions.add(new ExchangeCarrots(-10));
561      possibleMove.add(new Move(actions));
562      actions.clear();
563    }
564    if (GameRuleLogic.isValidToFallBack(this)) {
565      actions.add(new FallBack());
566      possibleMove.add(new Move(actions));
567      actions.clear();
568    }
569    // Generiere mögliche Vorwärtszüge
570    for (int i = 1; i <= GameRuleLogic.calculateMoveableFields(this.getCurrentPlayer().getCarrots()); i++) {
571      GameState clone = null;
572      try {
573        clone = this.clone();
574      } catch (CloneNotSupportedException e) {
575        e.printStackTrace();
576      }
577      // Überrüfe ob Vorwärtszug möglich ist
578      if (GameRuleLogic.isValidToAdvance(clone, i)) {
579        Advance tryAdvance = new Advance(i);
580        try {
581          tryAdvance.perform(clone);
582        } catch (InvalidMoveException e) {
583          // Sollte nicht passieren, da Zug valide ist
584          e.printStackTrace();
585          break;
586        }
587        actions.add(tryAdvance);
588        // überprüfe, ob eine Karte gespielt werden muss/kann
589        if (clone != null && clone.getCurrentPlayer().mustPlayCard()) {
590          possibleMove.addAll(clone.checkForPlayableCards(actions));
591        } else {
592          // Füge möglichen Vorwärtszug hinzu
593          possibleMove.add(new Move(actions));
594        }
595      }
596      actions.clear();
597    }
598    if (possibleMove.isEmpty()) {
599      Move move;
600      logger.warn("Muss aussetzen");
601      actions.add(new Skip());
602      move = new Move(actions);
603      possibleMove.add(move);
604    }
605    return possibleMove;
606  }

608  /**
609   * Überprüft für übergebenen GameState und bisher getätigte Züge,
610   * ob das Ausspielen einer Karte nötig/möglich ist
611   * @param actions bisherige Aktionenliste
612   * @return mögliche Züge
613   */
614  private ArrayList<Move> checkForPlayableCards(ArrayList<Action> actions) {
615    ArrayList<Move> possibleMove = new ArrayList<>();
616    if (this.getCurrentPlayer().mustPlayCard()) { // überprüfe, ob eine Karte gespielt werden muss
617      if (GameRuleLogic.isValidToPlayEatSalad(this)) {
618        actions.add(new Card(CardType.EAT_SALAD, actions.size()));
619        possibleMove.add(new Move(actions));
620
621        actions.remove(new Card(CardType.EAT_SALAD, 1));
622      }
623      if (GameRuleLogic.isValidToPlayTakeOrDropCarrots(this, 20)) {
624        actions.add(new Card(CardType.TAKE_OR_DROP_CARROTS, 20,  actions.size()));
625        possibleMove.add(new Move(actions));
626
627        actions.remove(new Card(CardType.TAKE_OR_DROP_CARROTS, 20, actions.size()));
628      }
629      if (GameRuleLogic.isValidToPlayTakeOrDropCarrots(this, -20)) {
630        actions.add(new Card(CardType.TAKE_OR_DROP_CARROTS, -20,  actions.size()));
631        possibleMove.add(new Move(actions));
632
633        actions.remove(new Card(CardType.TAKE_OR_DROP_CARROTS, -20, actions.size()));
634      }
635      if (GameRuleLogic.isValidToPlayTakeOrDropCarrots(this, 0)) {
636        actions.add(new Card(CardType.TAKE_OR_DROP_CARROTS, 0,  actions.size()));
637        possibleMove.add(new Move(actions));
638
639        actions.remove(new Card(CardType.TAKE_OR_DROP_CARROTS, 0,  actions.size()));
640      }
641      if (GameRuleLogic.isValidToPlayHurryAhead(this)) {
642        Card card = new Card(CardType.HURRY_AHEAD,  actions.size());
643        actions.add(card);
644        // Überprüfe ob wieder auf Hasenfeld gelandet:
645        GameState clone = null;
646        try {
647          clone = this.clone();
648        } catch (CloneNotSupportedException e) {
649          e.printStackTrace();
650        }
651        try {
652          card.perform(clone);
653        } catch (InvalidMoveException e) {
654          // Sollte nie passieren
655          e.printStackTrace();
656        }
657        if (clone != null && clone.getCurrentPlayer().mustPlayCard()) {
658          ArrayList<Move> moves = clone.checkForPlayableCards(actions);
659          if (!moves.isEmpty()) {
660            possibleMove.addAll(moves);
661          }
662        } else {
663          possibleMove.add(new Move(actions));
664        }
665
666        actions.remove(new Card(CardType.HURRY_AHEAD,  actions.size()));
667      }
668      if (GameRuleLogic.isValidToPlayFallBack(this)) {
669        Card card = new Card(CardType.FALL_BACK,  actions.size());
670        actions.add(card);
671        // Überprüfe ob wieder auf Hasenfeld gelandet:
672        GameState clone = null;
673        try {
674          clone = this.clone();
675        } catch (CloneNotSupportedException e) {
676          e.printStackTrace();
677        }
678        try {
679          card.perform(clone);
680        } catch (InvalidMoveException e) {
681          // Sollte nie passieren
682          e.printStackTrace();
683        }
684        if (clone != null && clone.getCurrentPlayer().mustPlayCard()) {
685          ArrayList<Move> moves = clone.checkForPlayableCards(actions);
686          if (!moves.isEmpty()) {
687            possibleMove.addAll(moves);
688          }
689        } else {
690          possibleMove.add(new Move(actions));
691        }
692        actions.remove(new Card(CardType.FALL_BACK,  actions.size()));
693      }
694    }
695    return possibleMove;
696  }
