006/**
007 * Ein Feld des Spielfelds. Ein Spielfeld ist durch den index eindeutig identifiziert.
008 * Das type Attribut gibt an, um welchen Feldtyp es sich handelt.
009 */
010@XStreamAlias(value = "field")
011public class Field {
012
013  @XStreamAsAttribute
014  private int index;
015  
016  @XStreamAsAttribute
017  private FieldType type;
018  
019  public Field(FieldType type) {
020    this(type, -1);
021  }
022  
023  private Field(FieldType type, int index) {
024    this.index = index;
025    this.type = type;
026  }
027  
028  public int getIndex() {
029    return index;
030  }
031
032  /**

034   * @param index Index des Feldes
035   */
036  public void setIndex(int index) {
037    this.index = index;
038  }
039
040  public FieldType getType() {
041    return type;
042  }
043
044  /**
045   * Nur f端r den Server (f端r Test) relevant.
046   * @param type Feldtyp
047   */
048  public void setType(FieldType type) {
049    this.type = type;
050  }
051  
052  @Override
053  public Field clone() {
054    return new Field(this.type, this.index);
055  }
056  
057}
