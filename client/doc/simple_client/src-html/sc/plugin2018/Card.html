001package sc.plugin2018;
002
003import com.thoughtworks.xstream.annotations.XStreamAlias;
004import com.thoughtworks.xstream.annotations.XStreamAsAttribute;
005import sc.plugin2018.util.GameRuleLogic;
006import sc.shared.InvalidMoveException;
007
008/**
009 * Eine Karte die auf einem Hasenfeld gespielt werden kann.
010 */
011@XStreamAlias(value = "card")
012public class Card extends Action {
013
014  @XStreamAsAttribute
015  private CardType type;
016
017  /**
018   * Nur für TAKE_OR_DROP_CARROTS genutzt. Muss 20, 0 oder -20 sein.
019   */
020  @XStreamAsAttribute
021  private int value;
022
023  /**
024   * Default Konstruktor. Setzt value auf 0 und order auf 0.
025   * @param type Art der Karte
026   */
027  public Card(CardType type) {
028    this.order = 0;
029    this.value = 0;
030    this.type = type;
031  }
032
033  /**
034   * Konstruktor für eine Karte
035   * @param type Art der Karte
036   * @param order Index in der Aktionsliste des Zuges
037   */
038  public Card(CardType type, int order) {
039    this.order = order;
040    this.value = 0;
041    this.type = type;
042  }
043
044  /**
045   *
046   * @param type Art der Karte
047   * @param value Wert einer Karte nur für TAKE_OR_DROP_CARROTS genutzt (-20,0,20)
048   * @param order Index in der Aktionsliste des Zuges
049   */
050  public Card(CardType type, int value, int order) {
051    this.order = order;
052    this.value = 0; // default value
053    this.value = value;
054    this.type = type;
055  }
056
057
058  @Override
059  public void perform(GameState state) throws InvalidMoveException {
060    state.getCurrentPlayer().setMustPlayCard(false); // player played a card
061    switch (type) { // when entering a HARE field with fall_back or hurry ahead, player has to play another card
062      case EAT_SALAD:
063        if (GameRuleLogic.isValidToPlayEatSalad(state)) {
064          state.getCurrentPlayer().eatSalad();
065          if (state.isFirst(state.getCurrentPlayer())) {
066            state.getCurrentPlayer().changeCarrotsBy(10);
067          } else {
068            state.getCurrentPlayer().changeCarrotsBy(30);
069          }
070        } else {
071          throw new InvalidMoveException("Das Ausspielen der EAT_SALAD Karte ist nicht möglich.");
072        }
073        break;
074      case FALL_BACK:
075        if (GameRuleLogic.isValidToPlayFallBack(state)) {
076          state.getCurrentPlayer().setFieldIndex(state.getOtherPlayer().getFieldIndex() - 1);
077          if (state.fieldOfCurrentPlayer() == FieldType.HARE) {
078            state.getCurrentPlayer().setMustPlayCard(true);
079          }
080        } else {
081          throw new InvalidMoveException("Das Ausspielen der FALL_BACK Karte ist nicht möglich.");
082        }
083        break;
084      case HURRY_AHEAD:
085        if (GameRuleLogic.isValidToPlayHurryAhead(state)) {
086          state.getCurrentPlayer().setFieldIndex(state.getOtherPlayer().getFieldIndex() + 1);
087          if (state.fieldOfCurrentPlayer() == FieldType.HARE) {
088            state.getCurrentPlayer().setMustPlayCard(true);
089          }
090        } else {
091          throw new InvalidMoveException("Das Ausspielen der HURRY_AHEAD Karte ist nicht möglich.");
092        }
093        break;
094      case TAKE_OR_DROP_CARROTS:
095        if (GameRuleLogic.isValidToPlayTakeOrDropCarrots(state, this.getValue())) {
096          state.getCurrentPlayer().changeCarrotsBy(this.getValue());
097        } else {
098          throw new InvalidMoveException("Das Ausspielen der TAKE_OR_DROP_CARROTS Karte ist nicht möglich.");
099        }
100        break;
101    }
102    state.setLastAction(this);
103    // remove player card
104    state.getCurrentPlayer().setCards(state.getCurrentPlayer().getCardsWithout(this.type));
105  }
106
107  public CardType getType() {
108    return type;
109  }
110
111  public int getValue() {
112    return value;
113  }
114
115  @Override
116  public Card clone() {
117    return new Card(this.type, this.value, this.order);
118  }
119
120  @Override
121  public boolean equals(Object o) {
122      return o instanceof Card && (this.value == ((Card) o).value) && (this.type == ((Card) o).type);
123  }
124
125  @Override
126  public String toString() {
127    return "Card "
128            + this.getType()
129            + ((this.getType() == CardType.TAKE_OR_DROP_CARROTS)?(" " + this.value):"")
130            + " order " + this.order;
131  }
132}
